	kolczyk ID matki(5)

Create table JALOWKI
(
	numer ID(14) primary key,
	kolczyk(5),
	data urodzenia(8),
	numer ID matki(14),
	kolczyk ID matki(5)

Drop sequence seq_hodowla_nr;
Drop sequence seq_byki_nr;
Drop sequence seq_jalowki_nr;
create sequence seq_hodowla_nr;
create sequence seq_byki_nr;
create sequence seq_jalowki_nr;
create or replace trigger tgr_hodowla_nr
before insert on HODOWLA
for each now
begin
IF :new.NR IS NULL THEN
		SELECT seq_hodowla_nr.nextval
    INTO :new.NR FROM DUAL;
	END IF;
END;
/
create or replace TRIGGER TGR_BYKI_NR 
BEFORE INSERT ON BYKI 
FOR EACH ROW
BEGIN
  IF :new.NR IS NULL THEN
		SELECT seq_byki_nr.nextval
    INTO :new.NR FROM DUAL;
	END IF;
END;
/
create or replace TRIGGER TGR_JALOWKI_NR 
BEFORE INSERT ON JALOWKI 
FOR EACH ROW
BEGIN
  IF :new.NR IS NULL THEN
		SELECT seq_jalowki_nr.nextval
    INTO :new.NR FROM DUAL;
	END IF;
END;
/
COMMIT;
