Drop table HODOWLA cascade constraints;
Drop table BYKI cascade constraints;
Drop table JALOWKI cascade constraints;

Create table HODOWLA
(
	numer ID(14) primary key,
	data paszportu(8) not null,
	kolczyk ID(5),
	data urodzenia(8),
	plec(2),
	rasa(9),
	numer ID matki(14)
);

Create table BYKI
(
	numer ID(14) primary key,
	kolczyk ID(5),
	data urodzenia(8),
	numer ID matki(14)
);

Create table JALOWKI
(
	numer ID(14) primary key,
	kolczyk(5),
	data urodzenia(8),
	numer ID matki(14)
);
Drop sequence seq_hodowla_nr;
Drop sequence seq_byki_nr;
Drop sequence seq_jalowki_nr;
create sequence seq_hodowla_nr;
create sequence seq_byki_nr;
create sequence seq_jalowki_nr;
create or replace trigger tgr_hodowla_nr
before insert on HODOWLA
for each now
begin
IF :new.NR IS NULL THEN
		SELECT seq_hodowla_nr.nextval
    INTO :new.NR FROM DUAL;
	END IF;
END;
/
create or replace TRIGGER TGR_STANOWISKO_NR 
BEFORE INSERT ON STANOWISKO 
FOR EACH ROW
BEGIN
  IF :new.NR IS NULL THEN
		SELECT seq_byki_nr.nextval
    INTO :new.NR FROM DUAL;
	END IF;
END;
/
create or replace TRIGGER TGR_DZIAL_NR 
BEFORE INSERT ON DZIAL 
FOR EACH ROW
BEGIN
  IF :new.NR IS NULL THEN
		SELECT seq_jalowki_nr.nextval
    INTO :new.NR FROM DUAL;
	END IF;
END;
/
COMMIT;
